221  T Hermite(float time, const T& point1, const T& slope1, const T& point2, const T& slope2)
222  {
223  float tt = time * time;
224  float ttt = tt * time;
225 
226  T p2 = point2;
227  TrackHelpers::Neighborhood(point1, p2);
228 
229  float h1 = 2.0f * ttt - 3.0f * tt + 1.0f;
230  float h2 = -2.0f * ttt + 3.0f * tt;
231  float h3 = ttt - 2.0f * tt + time;
232  float h4 = ttt - tt;
233 
234  T result = point1 * h1 + p2 * h2 + slope1 * h3 + slope2 * h4;
235  return TrackHelpers::AdjustHermiteResult(result);
